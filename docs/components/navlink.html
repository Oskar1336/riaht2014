    The purpose of this component is to make sure the

  • link element gets the active class as expected by bootstrap. See explanation here

var React = require('react');
var Link = require('react-router').Link;
var ActiveState = require('react-router').ActiveState;

var Navlink = React.createClass({
  mixins: [ ActiveState ],
  render: function() {
    var className = this.isActive(this.props.to, this.props.params, this.props.query) ? 'active' : '';
    return <li className={className}>{Link(this.props)}</li>;
  }
});

module.exports = Navlink;
